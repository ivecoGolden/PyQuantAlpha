```python
class Strategy:
    def init(self):
        # 初始化技术指标
        self.ema20 = EMA(20)
        self.ema60 = EMA(60)
        self.rsi = RSI(14)
        self.atr = ATR(14)
        self.bb = BollingerBands(20, 2.0)
        
        # 设置初始资金（可选，默认100,000）
        self.set_capital(100000)
        
        # 设置仓位管理：基于ATR风险控制，每次最大风险2%
        self.setsizer("risk", risk_percent=2, atr_multiplier=2)
        
        # 状态变量，用于避免重复入场
        self.in_position = False
    
    def on_bar(self, data):
        # 获取当前Bar数据（单资产模式，data是Bar对象）
        bar = data
        close = bar.close
        high = bar.high
        low = bar.low
        
        # 更新指标
        ema20_val = self.ema20.update(close)
        ema60_val = self.ema60.update(close)
        rsi_val = self.rsi.update(close)
        atr_val = self.atr.update(high, low, close)
        bb_val = self.bb.update(close)
        
        # 检查指标是否有效（避免初始阶段None值）
        if ema20_val is None or ema60_val is None or rsi_val is None or atr_val is None or bb_val is None:
            return
        
        # 获取当前持仓
        pos = self.get_position("BTCUSDT")
        self.in_position = pos is not None and pos.quantity != 0
        
        # 如果没有持仓，检查入场条件
        if not self.in_position:
            # 做多条件：EMA20 > EMA60（趋势向上）且 RSI < 40（超卖）且价格触及布林带下轨
            if ema20_val > ema60_val and rsi_val < 40 and close <= bb_val.lower:
                # 计算止损和止盈价格
                stop_price = close - 2 * atr_val
                take_profit = close + 4 * atr_val
                # 使用挂钩订单入场（自动设置止损和止盈）
                self.buy_bracket("BTCUSDT", stopprice=stop_price, limitprice=take_profit)
                print(f"做多入场: 价格={close}, 止损={stop_price:.2f}, 止盈={take_profit:.2f}")
            
            # 做空条件：EMA20 < EMA60（趋势向下）且 RSI > 60（超买）且价格触及布林带上轨
            elif ema20_val < ema60_val and rsi_val > 60 and close >= bb_val.upper:
                # 计算止损和止盈价格
                stop_price = close + 2 * atr_val
                take_profit = close - 4 * atr_val
                # 使用挂钩订单入场（自动设置止损和止盈）
                self.sell_bracket("BTCUSDT", stopprice=stop_price, limitprice=take_profit)
                print(f"做空入场: 价格={close}, 止损={stop_price:.2f}, 止盈={take_profit:.2f}")
    
    def notify_order(self, order):
        # 打印订单状态信息
        if order.status == "FILLED":
            print(f"订单成交: {order.id} {order.side} {order.quantity} @ {order.price}")
        elif order.status == "REJECTED":
            print(f"订单被拒绝: {order.id}")
        elif order.status == "CANCELLED":
            print(f"订单取消: {order.id}")
    
    def notify_trade(self, trade):
        # 打印交易盈亏信息
        if trade.pnl != 0:
            print(f"交易平仓: {trade.symbol} 盈亏={trade.pnl:.2f}")
```
